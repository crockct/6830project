SELECT d_labitems.test_name,
d_labitems.itemid as e_itemid,
COUNT(labevents.itemid) AS NumberOfOccurences
FROM labevents
JOIN d_labitems
ON (labevents.itemid=d_labitems.itemid)
WHERE fluid='BLOOD'
GROUP BY d_labitems.test_name
ORDER BY NumberOfOccurences DESC
LIMIT 5
